    if n <= 1 { return n }
    if n in memo { return memo[n] }
    memo[n] = fib(n-1) + fib(n-2)
    return memo[n]

#054Quantum Teleportationquantum▾
Transfer a quantum state from qubit 0 to qubit 2 using entanglement and classical communication.
simulate {
    let q = quantum_register(3)
    # Prepare state to teleport
    Ry(q[0], PI/3)
    # Create Bell pair between q[1] and q[2]
    H(q[1]); CNOT(q[1], q[2])
    # Bell measurement on q[0], q[1]
    CNOT(q[0], q[1]); H(q[0])
    let b0 = measure(q[0])
    let b1 = measure(q[1])
    # Classical corrections
    if b1 == 1 { X(q[2]) }
    if b0 == 1 { Z(q[2]) }
    print("State teleported to qubit 2!")
}

#063Deutsch-Jozsa Algorithmquantumalgorithm▾
Determine if a function is constant or balanced in ONE query. Classically needs 2^(n-1)+1 queries.
simulate {
    let n = 4
    let q = quantum_register(n + 1)
    X(q[n]); H(q[n])     # ancilla in |−⟩
    for i in range(n) { H(q[i]) }
    # Balanced oracle: CNOT from each qubit to ancilla
    for i in range(n) { CNOT(q[i], q[n]) }
    for i in range(n) { H(q[i]) }
    let r = measure_all(q, shots=100)
    print("Result:", r)  # non-zero → balanced
}
